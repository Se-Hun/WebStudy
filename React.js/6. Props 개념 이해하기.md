## 여러개의 컴포넌트에 데이터 전달하기

**<App.js>**

    import React, { Component } from 'react';
    import './App.css';
    import Movie from './Movie';
    
    const movies = [
    "Matrix",
    "Full Metal Jacket",
    "Oldboy",
    "Star wars"
    ]
    
    class App extends Component {
      render() {
        return (
          <div className="App">
            <Movie title={movies[0]}/>
            <Movie title={movies[1]}/>
            <Movie title={movies[2]}/>
            <Movie title={movies[3]}/>
          </div>
        );
      }
    }
    
    export default App;

배열과 각각의 컴포넌트에 배열의 인자들을 전달해주었다.
